using Android.App;
using Android.Content;
using Android.OS;
using Android.Runtime;
using Android.Views;
using Android.Widget;
using Java.Util;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace App3
{
    [Activity(Label = "ActivityDB", MainLauncher = true)]
    public class ActivityDB : Activity
    {
        EditText editText1;
        EditText editText2;
        Button button1;
        ListView listView1;
        List<City> cities;
        DBCities dBCities;
        protected override void OnCreate(Bundle savedInstanceState)
        {
            base.OnCreate(savedInstanceState);

            SetContentView(Resource.Layout.layout3);
            listView1 = FindViewById<ListView>(Resource.Id.listView1);
            button1 = FindViewById<Button>(Resource.Id.button1);
            button1.Click += Button1_Click;
            editText1 = FindViewById<EditText>(Resource.Id.editText1);
            editText2 = FindViewById<EditText>(Resource.Id.editText2);
            cities = new List<City>();
            NewMethod();
        }
        private void Button1_Click(object sender, EventArgs e)
        {
            dBCities = new DBCities(ApplicationContext);
            dBCities.Insert(editText1.Text, editText2.Text);
            NewMethod();
        }

        private void NewMethod()
        {
            cities.Clear();

            dBCities = new DBCities(ApplicationContext);
            var c = dBCities.Select();

            while (c.MoveToNext())
            {
                City city = new City
                {
                    Cityname = c.GetString(c.GetColumnIndex("Cityname")),
                    Population = c.GetInt(c.GetColumnIndex("Population")),
                };
                cities.Add(city);
            }

            listView1.Adapter = new CityAdapter(this, Resource.Layout.main_list_row, cities);
        }
    }
}
